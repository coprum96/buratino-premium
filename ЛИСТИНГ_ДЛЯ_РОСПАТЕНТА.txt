Программа "Золотой Детектор" представляет собой интерактивное 
образовательное веб-приложение, разработанное для обучения 
пользователей навыкам распознавания и защиты от финансового 
мошенничества.

КЛЮЧЕВЫЕ ОСОБЕННОСТИ:

1. Интерактивная образовательная система
   - 8 обучающих глав с последовательным прохождением
   - Система диалогов с эффектом печатной машинки
   - Интерактивные выборы с последствиями

2. Система тестирования знаний
   - Автоматизированное тестирование после каждой главы
   - Немедленная обратная связь
   - Образовательные объяснения к ответам

3. Счетчик жертв мошенничества
   - Реал-тайм статистика (старт: 13,495,302)
   - Автоматический инкремент каждые 2.5 секунды
   - Социальное доказательство важности проблемы

4. Образовательные материалы
   - 10 контрольных вопросов "Золотого Детектора"
   - База "красных флажков" мошенничества
   - Справочные материалы по финансовой грамотности

5. Геймификация обучения

   - Прогрессивное открытие глав

1. АЛГОРИТМ УПРАВЛЕНИЯ СОСТОЯНИЕМ (Zustand Store)

   Инициализация:
   - coins = 5 (игровая валюта)
   - wisdom = 0 (уровень мудрости 0-100)
   - currentLevel = 0 (текущая глава)
   - victimCount = 13495302 (счетчик жертв)
   - phase = 'landing' (текущий экран)
   
   Операции:
   - addCoins(amount): coins = max(0, coins + amount)
   - addWisdom(amount): wisdom = max(0, min(100, wisdom + amount))
   - nextLevel(): currentLevel++, currentDialogue = 0
   - completeLevel(id): completedLevels.push(id)

2. АЛГОРИТМ СЧЕТЧИКА ЖЕРТВ

   Глобальный таймер в App.tsx:
   
   setInterval(() => {
     victimCount = victimCount + 1;
   }, 2500); // каждые 2.5 секунды
   
   Форматирование: victimCount.toLocaleString('ru-RU')
   Результат: "13 495 302"

3. АЛГОРИТМ ЭФФЕКТА ПЕЧАТНОЙ МАШИНКИ

   Анимация текста в DialogueScene:
   
   FOR each symbol in text:
     displayedText = text[0...currentIndex]
     currentIndex++
     WAIT 30ms
   END FOR
   
   Результат: постепенное появление текста символ за символом

4. АЛГОРИТМ СИСТЕМЫ ТЕСТИРОВАНИЯ

   Процесс тестирования в QuizScreen:
   
   FOR each question in quiz:
     DISPLAY question with 4 options
     WAIT for user selection
     
     IF answer == correct:
       score++
       addWisdom(reward)
       SHOW green indicator
     ELSE:
       SHOW red indicator
     END IF
     
     SHOW explanation
     WAIT 3 seconds
     NEXT question
   END FOR
   
   percentage = (score / total) * 100
   
   DISPLAY result:
     >= 80% : "Отлично!"
     >= 60% : "Хорошо!"
     < 60%  : "Продолжай учиться!"

5. АЛГОРИТМ ПРОГРЕССИИ ГЛАВ

   Система открытия глав в ChapterMap:
   
   FOR each level in levels:
     isAvailable = (level.index <= currentLevel)
     isCompleted = (level.id IN completedLevels)
     isCurrent = (level.index == currentLevel)
     
     IF isAvailable:
       ALLOW click
       SHOW colored background
     ELSE:
       SHOW lock icon
       DISABLE click
     END IF
     
     IF isCompleted:
       SHOW checkmark ✅
     END IF
     
     IF isCurrent AND NOT isCompleted:
       SHOW star ⭐
       HIGHLIGHT with ring
     END IF
   END FOR

interface GameState {
  // Статистика игрока
  coins: number;                    // Игровая валюта
  wisdom: number;                   // Уровень мудрости (0-100)
  
  // Прогресс
  currentLevel: number;             // Текущая глава (0-7)
  currentDialogue: number;          // Текущий диалог в главе
  completedLevels: number[];        // Завершенные главы
  achievements: string[];           // Достижения
  
  // Тесты
  preTestScore: number;             // Результат предтеста
  postTestScore: number;            // Результат пост-теста
  
  // UI
  phase: GamePhase;                 // Текущий экран
  showMaterialsPanel: boolean;      // Показать справку
  
  // Счетчики
  globalCitizensCount: number;      // Счетчик граждан
  victimCount: number;              // Счетчик жертв мошенничества
  
  // Методы
  setPhase: (phase: GamePhase) => void;
  addCoins: (amount: number) => void;
  addWisdom: (amount: number) => void;
  nextLevel: () => void;
  setCurrentDialogue: (index: number) => void;
  completeLevel: (levelId: number) => void;
  addAchievement: (achievement: string) => void;
  toggleMaterialsPanel: () => void;
  setPostTestScore: (score: number) => void;
  resetGame: () => void;
  incrementCitizensCount: () => void;
}

type GamePhase = 
  | 'landing'       // Стартовый экран
  | 'chapterMap'    // Карта глав
  | 'dialogue'      // Сцена диалога
  | 'quiz'          // Тест по главе
  | 'scenarioTest'  // Сценарный тест
  | 'postTest'      // Итоговый тест
  | 'ending';       // Финальный экран

interface Dialogue {
  character: string;              // ID персонажа
  text: string;                   // Текст диалога
  choices: Choice[];              // Варианты ответов
  redFlags?: string[];            // Красные флажки
  mood?: 'happy' | 'sad' | 'angry' | 'thoughtful';
}

interface Choice {
  text: string;                   // Текст варианта
  icon: string;                   // Иконка (emoji)
  wisdom?: number;                // Изменение мудрости
  coins?: number;                 // Изменение монет
  next?: number;                  // Следующий диалог
  nextLevel?: boolean;            // Переход на следующую главу
  nextPhase?: string;             // Переход на другую фазу
}

interface QuizQuestion {
  text: string;                   // Текст вопроса
  options: string[];              // Варианты ответов (4 шт)
  correct: number;                // Индекс правильного ответа
  explanation: string;            // Объяснение
  reward?: number;                // Награда за правильный ответ
}

interface Level {
  id: number;                     // Уникальный ID
  title: string;                  // Название главы
  subtitle: string;               // Подзаголовок
  gradient: string;               // CSS градиент
  dialogues: Dialogue[];          // Массив диалогов
  quiz: QuizQuestion[];           // Вопросы теста
  miniGame?: 'calculator' | 'baskets' | 'barometer' | 'contracts';
}

interface Character {
  id: string;                     // Уникальный ID
  name: string;                   // Имя персонажа
  emoji: string;                  // Emoji для отображения
  image?: string;                 // Путь к изображению
  description: string;            // Описание персонажа
}

interface EducationalMaterial {
  id: number;                     // ID материала
  title: string;                  // Заголовок
  sections: {                     // Секции материала
    name: string;                 // Название секции
    content: string;              // Содержание
  }[];
}

interface RedFlag {
  name: string;                   // Название флажка
  examples: string[];             // Примеры проявления
}

                    ЗАКЛЮЧЕНИЕ
════════════════════════════════════════════════════════════════

Представленный листинг демонстрирует ключевые компоненты и алгоритмы
образовательной системы "Золотой Детектор", предназначенной для 
обучения граждан Российской Федерации навыкам защиты от финансового
мошенничества.

Программа включает:
✓ Интерактивную обучающую систему с геймификацией
✓ Систему тестирования и оценки знаний
✓ Базу образовательных материалов
✓ Уникальный счетчик жертв мошенничества
✓ Методики распознавания мошеннических схем

Общий объем программы: 77 файлов, ~5 933 строк кода

Технологии: React, TypeScript, Zustand, Tailwind CSS, Vite

Дата составления листинга: 15 ноября 2024 года
